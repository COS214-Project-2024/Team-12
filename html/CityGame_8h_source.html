259  if (x1 == x2) { // Vertical path

266  grid.setTransportPath({x1, y}, '|');
267  }
268  } else if (y1 == y2) { // Horizontal path
269  for (int x = std::min(x1, x2) + 1; x < std::max(x1, x2); ++x) {

271  std::cout << "\033[1;31mNot enough money to complete transport link!\033[0m\n";

275  grid.setTransportPath({x, y1}, '-');
276  }
277  } else { // Diagonal or other complex path
278  int dx = (x2 > x1) ? 1 : -1;
279  int dy = (y2 > y1) ? 1 : -1;
280  int x = x1 + dx, y = y1 + dy;
281  while (x != x2 && y != y2) {
282  if (government.getMoney() < linkCost) {
283  std::cout << "\033[1;31mNot enough money to complete transport link!\033[0m\n";
284  return;
285  }

287  grid.setTransportPath({x, y}, '+');
288  x += dx;
